<DESCRIPTION>
This approach uses a fundamentally different algorithmic strategy based on adaptive pressure scoring rather than fixed threshold rules. Instead of hard-coded thresholds, it computes a dynamic "pressure score" that combines multiple signals with adaptive weights based on current system state. The key innovation is using exponential decay functions and relative change detection to capture system dynamics more effectively. It also employs a multi-stage decision process where different pressure levels trigger different rejection strategies, making it more sensitive to emerging congestion patterns while being less prone to false admits during transient spikes.
</DESCRIPTION>

def predict(features: Dict[str, float]) -> int:
    """
    Adaptive pressure scoring heuristic for SSD I/O admission control.
    
    Uses dynamic pressure computation with exponential decay and relative
    change detection to identify congestion patterns more effectively than
    fixed thresholds.

    Args:
        features: dict mapping feature name -> float value.

    Returns:
        1  →  REJECT  (predicted slow/high-latency I/O — block or hedge)
        0  →  KEEP    (predicted fast/normal I/O — let through)
    """
    # Extract core features
    latency = features["latency"]
    queue_len = features["queue_len"]
    size = features["size"]
    
    # Historical data
    prev_latencies = [
        features["prev_latency_1"],
        features["prev_latency_2"], 
        features["prev_latency_3"]
    ]
    prev_queues = [
        features["prev_queue_len_1"],
        features["prev_queue_len_2"],
        features["prev_queue_len_3"]
    ]
    prev_throughputs = [
        features["prev_throughput_1"],
        features["prev_throughput_2"],
        features["prev_throughput_3"]
    ]
    
    # Compute exponentially weighted moving averages (more recent = higher weight)
    weights = [0.5, 0.3, 0.2]  # Recent to old
    
    ewma_latency = sum(w * lat for w, lat in zip(weights, prev_latencies))
    ewma_queue = sum(w * q for w, q in zip(weights, prev_queues))
    ewma_throughput = sum(w * tp for w, tp in zip(weights, prev_throughputs))
    
    # Detect trends (positive = increasing pressure)
    latency_trend = (prev_latencies[0] - prev_latencies[2]) / max(prev_latencies[2], 1.0)
    queue_trend = (prev_queues[0] - prev_queues[2]) / max(prev_queues[2], 1.0)
    throughput_trend = (prev_throughputs[2] - prev_throughputs[0]) / max(prev_throughputs[0], 1.0)
    
    # Compute adaptive pressure score
    pressure_score = 0.0
    
    # Current state pressure (40% weight)
    current_pressure = 0.0
    if latency > 50:
        current_pressure += (latency - 50) / 200.0  # Normalize to [0,1] range
    if queue_len > 1:
        current_pressure += (queue_len - 1) / 15.0
    
    pressure_score += 0.4 * min(current_pressure, 1.0)
    
    # Historical pressure (30% weight)  
    hist_pressure = 0.0
    if ewma_latency > 30:
        hist_pressure += (ewma_latency - 30) / 150.0
    if ewma_queue > 1:
        hist_pressure += (ewma_queue - 1) / 12.0
    
    pressure_score += 0.3 * min(hist_pressure, 1.0)
    
    # Trend pressure (20% weight)
    trend_pressure = 0.0
    if latency_trend > 0.1:  # Latency increasing
        trend_pressure += min(latency_trend, 1.0) * 0.6
    if queue_trend > 0.1:  # Queue growing
        trend_pressure += min(queue_trend, 1.0) * 0.4
    if throughput_trend > 0.1:  # Throughput dropping
        trend_pressure += min(throughput_trend, 1.0) * 0.3
    
    pressure_score += 0.2 * min(trend_pressure, 1.0)
    
    # Size penalty (10% weight) - larger requests more likely to cause issues
    size_penalty = 0.0
    if size >= 32768:
        size_penalty = min((size - 32768) / 65536.0, 0.5)
    
    pressure_score += 0.1 * size_penalty
    
    # Multi-stage decision with adaptive thresholds
    # Stage 1: Critical pressure - immediate reject
    if pressure_score > 0.8:
        return 1
    
    # Stage 2: High pressure with size consideration
    if pressure_score > 0.6 and size >= 16384:
        return 1
        
    # Stage 3: Medium pressure with queue state
    if pressure_score > 0.4 and queue_len > 4:
        return 1
    
    # Stage 4: Low-medium pressure with latency spike protection
    if pressure_score > 0.3 and latency > 100:
        return 1
        
    # Stage 5: Emergency brake for extreme current conditions
    if latency > 300 or queue_len > 12:
        return 1
    
    return 0

def run_experiment(
    features: List[Dict[str, float]],
    labels: List[int],
    seed: int = 42,
) -> Dict:
    """
    Run the heuristic over the provided feature list and return predictions.
    The evaluator compares predictions against labels to compute metrics.
    """
    predictions = [predict(f) for f in features]
    return {
        "predictions": predictions,
        "labels":      labels,
    }
