from typing import Dict, List

# EVOLVE-BLOCK-START
def predict(features: Dict[str, float]) -> int:
    """
    Simple baseline heuristic for SSD I/O admission control.
    Returns 1 to REJECT, 0 to KEEP.
    """
    # Simple threshold-based approach
    latency = features.get("latency", 0)
    queue_len = features.get("queue_len", 0)
    
    # Reject if current latency is very high
    if latency > 300:
        return 1
    
    # Reject if queue is very congested
    if queue_len > 10:
        return 1
    
    # Otherwise keep
    return 0
# EVOLVE-BLOCK-END

def run_experiment(features: List[Dict[str, float]], labels: List[int], seed: int = 42) -> Dict:
    """Run the heuristic and return predictions."""
    predictions = [predict(f) for f in features]
    return {"predictions": predictions, "labels": labels}
